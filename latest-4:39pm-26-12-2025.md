// Pincode Locator JavaScript
document.addEventListener('DOMContentLoaded', function() {
    // DOM Elements
    const stateSelect = document.getElementById('stateSelect');
    const districtSelect = document.getElementById('districtSelect');
    const alphabetSelect = document.getElementById('alphabetSelect');
    const postOfficeSelect = document.getElementById('postOfficeSelect');
    const searchBtn = document.getElementById('searchBtn');
    const resetBtn = document.getElementById('resetBtn');
    const pincodeBadge = document.getElementById('pincodeBadge');
    const locationInfo = document.getElementById('locationInfo');
    const officeType = document.getElementById('officeType');
    const deliveryStatus = document.getElementById('deliveryStatus');
    const nearbyList = document.getElementById('nearbyList');
    const lastUpdated = document.getElementById('lastUpdated');
    
    let pincodeData = [];
    let selectedState = '';
    let selectedDistrict = '';
    let selectedAlphabet = '';
    
    // Initialize
    initPincodeLocator();
    
    async function initPincodeLocator() {
        try {
            showLoading(true);
            
            // Load data from GitHub
            const response = await fetch(pincodeLocator.githubDataUrl);
            const data = await response.json();
            
            pincodeData = data.pincodes || [];
            lastUpdated.textContent = data.metadata?.last_updated || '‚Äî';
            
            // Load states
            loadStates();
            
            // Setup event listeners
            setupEventListeners();
            
            showLoading(false);
        } catch (error) {
            console.error('Error loading pincode data:', error);
            showError('Failed to load pincode data. Please try again later.');
            showLoading(false);
        }
    }
    
    function loadStates() {
        // Get unique states
        const states = [...new Set(pincodeData.map(item => item.state))].sort();
        
        // Populate state dropdown
        stateSelect.innerHTML = '<option value="">Select State</option>';
        states.forEach(state => {
            const option = document.createElement('option');
            option.value = state;
            option.textContent = state;
            stateSelect.appendChild(option);
        });
    }
    
    function setupEventListeners() {
        // State selection
        stateSelect.addEventListener('change', function() {
            selectedState = this.value;
            districtSelect.disabled = !selectedState;
            alphabetSelect.disabled = true;
            postOfficeSelect.disabled = true;
            
            if (selectedState) {
                loadDistricts(selectedState);
            } else {
                clearDistrict();
                clearAlphabet();
                clearPostOffice();
            }
        });
        
        // District selection
        districtSelect.addEventListener('change', function() {
            selectedDistrict = this.value;
            alphabetSelect.disabled = !selectedDistrict;
            postOfficeSelect.disabled = true;
            
            if (selectedDistrict) {
                loadAlphabets(selectedState, selectedDistrict);
            } else {
                clearAlphabet();
                clearPostOffice();
            }
        });
        
        // Alphabet selection
        alphabetSelect.addEventListener('change', function() {
            selectedAlphabet = this.value;
            postOfficeSelect.disabled = !selectedAlphabet;
            
            if (selectedAlphabet) {
                loadPostOffices(selectedState, selectedDistrict, selectedAlphabet);
            } else {
                clearPostOffice();
            }
        });
        
        // Post office selection
        postOfficeSelect.addEventListener('change', function() {
            if (this.value) {
                showPincodeDetails(this.value);
            } else {
                clearResults();
            }
        });
        
        // Search button
        searchBtn.addEventListener('click', function() {
            if (postOfficeSelect.value) {
                showPincodeDetails(postOfficeSelect.value);
            } else {
                showError('Please select a post office first');
            }
        });
        
        // Reset button
        resetBtn.addEventListener('click', function() {
            resetAll();
        });
    }
    
    function loadDistricts(state) {
        // Get districts for selected state
        const districts = [...new Set(
            pincodeData
                .filter(item => item.state === state)
                .map(item => item.district)
        )].sort();
        
        // Populate district dropdown
        districtSelect.innerHTML = '<option value="">Select District</option>';
        districts.forEach(district => {
            const option = document.createElement('option');
            option.value = district;
            option.textContent = district;
            districtSelect.appendChild(option);
        });
        
        clearAlphabet();
        clearPostOffice();
        clearResults();
    }
    
    function loadAlphabets(state, district) {
        // Get first letters of post offices in selected district
        const alphabets = [...new Set(
            pincodeData
                .filter(item => item.state === state && item.district === district)
                .map(item => item.post_office.charAt(0).toUpperCase())
        )].sort();
        
        // Populate alphabet dropdown
        alphabetSelect.innerHTML = '<option value="">Select First Letter</option>';
        alphabets.forEach(letter => {
            const option = document.createElement('option');
            option.value = letter;
            option.textContent = letter;
            alphabetSelect.appendChild(option);
        });
        
        clearPostOffice();
        clearResults();
    }
    
    function loadPostOffices(state, district, alphabet) {
        // Get post offices starting with selected alphabet
        const postOffices = pincodeData
            .filter(item => 
                item.state === state && 
                item.district === district &&
                item.post_office.toUpperCase().startsWith(alphabet)
            )
            .sort((a, b) => a.post_office.localeCompare(b.post_office));
        
        // Populate post office dropdown
        postOfficeSelect.innerHTML = '<option value="">Select Post Office</option>';
        postOffices.forEach(office => {
            const option = document.createElement('option');
            option.value = office.post_office;
            option.textContent = office.post_office;
            option.dataset.pincode = office.pincode;
            postOfficeSelect.appendChild(option);
        });
        
        clearResults();
    }
    
    function showPincodeDetails(postOfficeName) {
        // Find the selected post office
        const selectedOffice = pincodeData.find(item => 
            item.post_office === postOfficeName
        );
        
        if (!selectedOffice) {
            showError('Post office not found');
            return;
        }
        
        // Update pincode badge
        pincodeBadge.textContent = selectedOffice.pincode;
        
        // Update location info
        locationInfo.innerHTML = `
            <h4>${selectedOffice.post_office}</h4>
            <p>üìç ${selectedOffice.district}, ${selectedOffice.state}</p>
        `;
        
        // Update office details
        officeType.textContent = selectedOffice.office_type || '‚Äî';
        deliveryStatus.textContent = selectedOffice.delivery_status || '‚Äî';
        
        // Show nearby post offices
        showNearbyPostOffices(selectedOffice);
    }
    
    function showNearbyPostOffices(currentOffice) {
        // Find post offices in same district
        const nearby = pincodeData
            .filter(item => 
                item.district === currentOffice.district &&
                item.post_office !== currentOffice.post_office
            )
            .slice(0, 5); // Limit to 5
        
        // Update nearby list
        if (nearby.length > 0) {
            nearbyList.innerHTML = '';
            nearby.forEach(office => {
                const li = document.createElement('li');
                li.innerHTML = `
                    <strong>${office.post_office}</strong>
                    <span class="pincode">${office.pincode}</span>
                `;
                li.addEventListener('click', () => {
                    postOfficeSelect.value = office.post_office;
                    showPincodeDetails(office.post_office);
                });
                nearbyList.appendChild(li);
            });
        } else {
            nearbyList.innerHTML = '<li class="empty-state">No nearby post offices found</li>';
        }
    }
    
    function clearDistrict() {
        districtSelect.innerHTML = '<option value="">Select District</option>';
        districtSelect.disabled = true;
    }
    
    function clearAlphabet() {
        alphabetSelect.innerHTML = '<option value="">Select First Letter</option>';
        alphabetSelect.disabled = true;
    }
    
    function clearPostOffice() {
        postOfficeSelect.innerHTML = '<option value="">Select Post Office</option>';
        postOfficeSelect.disabled = true;
    }
    
    function clearResults() {
        pincodeBadge.textContent = '‚Äî ‚Äî ‚Äî ‚Äî ‚Äî ‚Äî';
        locationInfo.innerHTML = '<p>Select a post office to see details</p>';
        officeType.textContent = '‚Äî';
        deliveryStatus.textContent = '‚Äî';
        nearbyList.innerHTML = '<li class="empty-state">Select a post office to see nearby locations</li>';
    }
    
    function resetAll() {
        stateSelect.value = '';
        clearDistrict();
        clearAlphabet();
        clearPostOffice();
        clearResults();
        
        // Reset disabled states
        districtSelect.disabled = true;
        alphabetSelect.disabled = true;
        postOfficeSelect.disabled = true;
    }
    
    function showLoading(isLoading) {
        const container = document.querySelector('.pincode-locator-container');
        if (isLoading) {
            container.classList.add('loading');
        } else {
            container.classList.remove('loading');
        }
    }
    
    function showError(message) {
        // Create error message element
        const errorDiv = document.createElement('div');
        errorDiv.className = 'error-message';
        errorDiv.textContent = message;
        
        // Insert after header
        const header = document.querySelector('.pincode-header');
        header.parentNode.insertBefore(errorDiv, header.nextSibling);
        
        // Remove after 5 seconds
        setTimeout(() => {
            if (errorDiv.parentNode) {
                errorDiv.parentNode.removeChild(errorDiv);
            }
        }, 5000);
    }
    
    function showSuccess(message) {
        // Create success message element
        const successDiv = document.createElement('div');
        successDiv.className = 'success-message';
        successDiv.textContent = message;
        
        // Insert after header
        const header = document.querySelector('.pincode-header');
        header.parentNode.insertBefore(successDiv, header.nextSibling);
        
        // Remove after 3 seconds
        setTimeout(() => {
            if (successDiv.parentNode) {
                successDiv.parentNode.removeChild(successDiv);
            }
        }, 3000);
    }
});

// data-loader.js
// This file loads and caches the GitHub data

class DataLoader {
    constructor(githubUrl) {
        this.githubUrl = githubUrl;
        this.cacheKey = 'pincode_data_cache';
        this.cacheExpiry = 24 * 60 * 60 * 1000; // 24 hours
    }
    
    async loadData() {
        // Try cache first
        const cached = this.getCachedData();
        if (cached) {
            console.log('Using cached data');
            return cached;
        }
        
        // Fetch from GitHub
        try {
            console.log('Fetching fresh data from GitHub');
            const response = await fetch(this.githubUrl);
            const data = await response.json();
            
            // Cache the data
            this.cacheData(data);
            
            return data;
        } catch (error) {
            console.error('Error loading data:', error);
            throw error;
        }
    }
    
    cacheData(data) {
        const cacheItem = {
            timestamp: Date.now(),
            data: data
        };
        localStorage.setItem(this.cacheKey, JSON.stringify(cacheItem));
    }
    
    getCachedData() {
        const cached = localStorage.getItem(this.cacheKey);
        if (!cached) return null;
        
        const cacheItem = JSON.parse(cached);
        const isExpired = Date.now() - cacheItem.timestamp > this.cacheExpiry;
        
        if (isExpired) {
            localStorage.removeItem(this.cacheKey);
            return null;
        }
        
        return cacheItem.data;
    }
    
    clearCache() {
        localStorage.removeItem(this.cacheKey);
    }
}

// Initialize with your GitHub URL
const dataLoader = new DataLoader('YOUR_GITHUB_RAW_URL_HERE');
export default dataLoader;
